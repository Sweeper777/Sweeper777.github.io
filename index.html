open class Gift {
    private var stuffInside: AnyObject

    open init(stuffInside: AnyObject) {
        self.stuffInside = stuffInside
    }

    open func opened() -> AnyObject {
        return stuffInside
    }

    open func send(to person: String) {
        print("I sent the present to \(person)")
    }
}
//--------
